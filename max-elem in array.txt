//to find the max element in array
#include<stdio.h>
void main(){
    int i,max,a[5];
    printf("enter arrays to continue\n");
    for(i=0;i<5;i++){
        scanf("%d", &a[i]);
    }
    max=0;
    for(i=0;i<5;i++){
        if(a[i]>max){
            max=a[i];
        }
    }
    printf("max array elements: %d", max);
}
