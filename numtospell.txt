#include<stdio.h>
void main(){
    int a;
    printf("enter number");
    scanf("%d", &a);
    if(1<=a && a<=5){
        if(a==1)
        printf("one");
        else if(a==2)
        printf("two");
        else if(a==3)
        printf("three");
        else if(a==4)
        printf("four");
        else if(a==5)
        printf("five");
    }
    else{
        printf("number is out of limit");
    }
}
